```kotlin
@OptIn(ExperimentalLayoutApi::class)
@Composable
fun ServiceChipList(
    services: List<InterpretedServiceDto>,
    maxItems: Int = Int.MAX_VALUE
) {
    if (services.isEmpty()) return
    FlowRow(
        modifier = Modifier.fillMaxWidth(),
        horizontalArrangement = Arrangement.spacedBy(Spacing.xs)
    ) {
        services.take(maxItems).forEach { service ->
            val isPrimary = service.matchLevel == "PRIMARY"
            SuggestionChip(
                onClick = {},
                label = { Text(service.displayName) },
                colors = if (isPrimary)
                    SuggestionChipDefaults.suggestionChipColors(
                        containerColor = MaterialTheme.colorScheme.primaryContainer,
                        labelColor = MaterialTheme.colorScheme.onPrimaryContainer,
                    )
                else
                    SuggestionChipDefaults.suggestionChipColors(),
            )
        }
    }
}
```

```kotlin
package com.fugisawa.quemfaz.ui.components

import androidx.compose.animation.core.FastOutSlowInEasing
import androidx.compose.animation.core.RepeatMode
import androidx.compose.animation.core.animateFloat
import androidx.compose.animation.core.infiniteRepeatable
import androidx.compose.animation.core.rememberInfiniteTransition
import androidx.compose.animation.core.tween
import androidx.compose.foundation.background
import androidx.compose.foundation.layout.Box
import androidx.compose.foundation.shape.RoundedCornerShape
import androidx.compose.material3.MaterialTheme
import androidx.compose.runtime.Composable
import androidx.compose.runtime.getValue
import androidx.compose.ui.Modifier
import androidx.compose.ui.draw.clip
import androidx.compose.ui.unit.dp

/**
 * A pulsing placeholder box used to indicate loading content (shimmer skeleton).
 * Adapts to the current theme's surface color — works correctly in both light and dark mode.
 */
@Composable
fun ShimmerBox(modifier: Modifier = Modifier) {
    val infiniteTransition = rememberInfiniteTransition(label = "shimmer")
    val alpha by infiniteTransition.animateFloat(
        initialValue = 0.35f,
        targetValue = 0.7f,
        animationSpec = infiniteRepeatable(
            animation = tween(durationMillis = 800, easing = FastOutSlowInEasing),
            repeatMode = RepeatMode.Reverse,
        ),
        label = "shimmerAlpha",
    )
    Box(
        modifier = modifier
            .clip(RoundedCornerShape(4.dp))
            .background(MaterialTheme.colorScheme.surfaceVariant.copy(alpha = alpha))
    )
}
```

### Task 6: AnimatedContent Transitions in OnboardingScreens

**Files:**
- Modify: `composeApp/src/commonMain/kotlin/com/fugisawa/quemfaz/screens/OnboardingScreens.kt`

Currently the `when (uiState)` block inside `OnboardingScreens` switches branches with no animation. Wrap it in `AnimatedContent` so each state transition slides in from the right (forward) or fades (for Loading/Error).

The `Loading` and `Error` states should use a fade; step transitions (Idle → NeedsClarification → DraftReady → PhotoRequired → KnownName) should use a horizontal slide.

- [ ] **Step 0: Verify `compose.animation` is available**

The `AnimatedContent`, `slideInHorizontally`, `slideOutHorizontally`, `fadeIn`, `fadeOut`, and `togetherWith` APIs live in `androidx.compose.animation`. This library is transitively pulled in by `compose.foundation` (which is already a dependency). The project compiles today with animations inside `ModalBottomSheet`, confirming it's available.

**If the build fails with unresolved references for animation APIs:** add an explicit dependency. Check `gradle/libs.versions.toml` for an existing `compose.animation` alias. If none exists, add to `commonMain.dependencies` in `build.gradle.kts`:
```kotlin
implementation("org.jetbrains.compose.animation:animation:${libs.versions.composeMultiplatform.get()}")
```
No change is needed proactively — only add if compilation fails.

> **Known UX limitation:** The slide direction is always "forward" (slide in from right, slide out to left) regardless of whether the user is navigating forward or backward. When `onBack` fires, the transition regresses (e.g., `DraftReady` → `Idle`) but still shows a forward slide. Implementing direction-aware transitions requires comparing step indices of `initialState` and `targetState` — this is out of scope for this task.

```kotlin
                initialState is OnboardingUiState.Loading ||
                initialState is OnboardingUiState.Error ||
                initialState is OnboardingUiState.Published

                fadeIn() togetherWith fadeOut()

            }
```
