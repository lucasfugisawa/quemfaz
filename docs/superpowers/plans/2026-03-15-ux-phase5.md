# UX Phase 5 — Marketplace Signals & Performance Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Feed engagement signals into search ranking, display granular recency, add client-side search caching, configure Ktor timeouts, and add Cache-Control headers.

**Architecture:** Server-side changes add denormalized engagement counters to `professional_profiles`, wire them into the ranking algorithm, and compute `daysSinceActive` in all three `mapToResponse()` functions. Client-side changes add in-memory search caching in `HomeViewModel`, granular recency display in `StatusChipRow`, and `HttpTimeout` in `ApiClient`.

**Tech Stack:** Kotlin, Ktor (server + client), Exposed ORM, Flyway, Compose Multiplatform, kotlinx-datetime, Koin DI

**Spec:** `docs/superpowers/specs/2026-03-15-ux-phase5-design.md`

    private fun createProfile(
        id: String = "p1",
        serviceIds: List<Pair<String, ServiceMatchLevel>> = emptyList(),
        viewCount: Int = 0,

        status = ProfessionalProfileStatus.PUBLISHED,
        lastActiveAt = lastActiveAt,
        createdAt = Instant.now(),
        updatedAt = Instant.now(),
        viewCount = viewCount,
        contactClickCount = contactClickCount,
    )

    private fun createQuery(
        serviceIds: List<String> = listOf("svc1"),
        cityName: String? = "São Paulo",
        neighborhoods: List<String> = emptyList(),
    ) = InterpretedSearchQuery(
        originalQuery = "test",
        normalizedQuery = "test",
        serviceIds = serviceIds,
        cityName = cityName,
        neighborhoods = neighborhoods,
        freeTextAliases = emptyList(),
    )

    @Test
    fun `profile with contact clicks ranks higher than one without`() {
        val profileWithClicks = createProfile(
            id = "with-clicks",
            serviceIds = listOf("svc1" to ServiceMatchLevel.PRIMARY),
            contactClickCount = 10,
        )
        val profileWithout = createProfile(
            id = "without-clicks",
            serviceIds = listOf("svc1" to ServiceMatchLevel.PRIMARY),
            contactClickCount = 0,
        )
        val ranked = service.rank(listOf(profileWithout, profileWithClicks), createQuery())
        assertTrue(ranked.first().id.value == "with-clicks")
    }

    @Test
    fun `profile with views ranks higher than one without`() {
        val profileWithViews = createProfile(
            id = "with-views",
            serviceIds = listOf("svc1" to ServiceMatchLevel.PRIMARY),
            viewCount = 30,
        )
        val profileWithout = createProfile(
            id = "without-views",
            serviceIds = listOf("svc1" to ServiceMatchLevel.PRIMARY),
            viewCount = 0,
        )
        val ranked = service.rank(listOf(profileWithout, profileWithViews), createQuery())
        assertTrue(ranked.first().id.value == "with-views")
    }

    @Test
    fun `contact click score is capped at 20 clicks`() {
        val profileAtCap = createProfile(
            id = "at-cap",
            serviceIds = listOf("svc1" to ServiceMatchLevel.PRIMARY),
            contactClickCount = 20,
        )
        val profileOverCap = createProfile(
            id = "over-cap",
            serviceIds = listOf("svc1" to ServiceMatchLevel.PRIMARY),
            contactClickCount = 100,
        )
        // Profile over cap should NOT rank higher than profile at cap
        // Both get min(clicks, 20) * 2 = 40 points from clicks
        val rankedOverFirst = service.rank(listOf(profileOverCap, profileAtCap), createQuery())
        val rankedAtFirst = service.rank(listOf(profileAtCap, profileOverCap), createQuery())
        // With stable sort and equal scores, input order is preserved both ways
        assertTrue(rankedOverFirst[0].id.value == "over-cap")
        assertTrue(rankedAtFirst[0].id.value == "at-cap")
    }

    @Test
    fun `view score is capped at 50 views`() {
        val profileAtCap = createProfile(
            id = "at-cap",
            serviceIds = listOf("svc1" to ServiceMatchLevel.PRIMARY),
            viewCount = 50,
        )
        val profileOverCap = createProfile(
            id = "over-cap",
            serviceIds = listOf("svc1" to ServiceMatchLevel.PRIMARY),
            viewCount = 200,
        )
        // Both get min(views, 50) * 0.5 = 25 points — equal scores
        val rankedOverFirst = service.rank(listOf(profileOverCap, profileAtCap), createQuery())
        val rankedAtFirst = service.rank(listOf(profileAtCap, profileOverCap), createQuery())
        assertTrue(rankedOverFirst[0].id.value == "over-cap")
        assertTrue(rankedAtFirst[0].id.value == "at-cap")
    }
}

    if (query.isBlank()) return

    // Check cache
    val cacheKey = "${query.lowercase()}:${currentCity.value?.lowercase()}"
    val cached = searchCache[cacheKey]
    val now = Clock.System.now().toEpochMilliseconds()
    if (cached != null && now - cached.timestamp < CACHE_TTL_MS) {

        lastQuery = query
        currentPage = cached.currentPage
        isLoadingMore = false
        _accumulatedResults.value = cached.accumulatedResults

        _searchUiState.value = SearchUiState.Success(
